CLS.JS

فریمورکی برای کار با استایل در ترمینال

نصب و استفاده

npm install --save cls.js

 style وارد کردن تابع 

const { style } = require("cls.js")

 Loc وارد کردن زمره 

const { Loc } = require("cls.js")

زمره همان کلاس به زبان پارسی است

style کار با تابع 

درباره تابع 

این تابع اصلی برای کار با این فریمورک است ، در این تابع مشخصه هایی جهت فرمان دادن به فریمورک وجود دارد 
که تمامی انها در ادامه به شما معرفی خواهند شد ، این مشخصه ها را تنها میتوانید به شکل شی استفاده کنید، ساده تر عرض کنم ، ورودی یک شی میگیرد که مشخصه های ان را به شما معرفی خواهیم کرد 

 text مشخصه  

const { style } = require("cls.js");
style({
 text : "متن من",
});

مشخصه تکست ورودی یک رشته دریافت میکند که ان را در خروجی نشان خواهد داد ، خروجی تنها متنی است که وارد میکنید

 color مشخصه  

const { style } = require("cls.js");
style({
 text : "متن من",
 color : "red"
});

 مشخصه رنگ تنها برای رنگ دادن به متن استفاده میشود و ورودی تنها میتوانید از متغیر های اماده یا ار جی بی باشد
 
در مشخصه رنگ که در بالا وارد کرده ایم خروجی جمله متن من با رنگ قرمز است شما میتوانید از متغیر های رنگی
 اماده همانند بالا استفاده بکنید یا از ویژگی ار جی بی استفاده کنید مثال آن را در کادر زیر میتوانید مشاهده کنید 

const { style } = require("cls.js");
style({
 text : "متن من",
 color : "rgb:255;0;0"
});

 bgColor مشخصه  

const { style } = require("cls.js");
style({
 text : "متن من",
 bgColor : "red"
});

 مشخصه رنگ پس زمینه تنها برای رنگ دادن به پس زمینه متن استفاده میشود و ورودی تنها میتوانید از متغیر های اماده یا ار جی بی باشد
 
در مشخصه رنگ پس زمینه که در بالا وارد کرده ایم خروجی جمله متن من با رنگ سفید و پس زمینه قرمز  است شما  میتوانید از متغیر های رنگی  اماده همانند بالا استفاده بکنید یا از ویژگی ار جی بی استفاده کنید مثال آن را در کادر زیر میتوانید مشاهده کنید 

const { style } = require("cls.js");
style({
 text : "متن من",
 bgColor : "rgb:255;0;0"
});

 font_style مشخصه  

const { style } = require("cls.js");
style({
 text : "متن من",
 font_style : "bold"
});

مشخصه فونت استایل ورودی یک رشته دریافت میکند که یک متغیر اماده است جهت تعیین نوع فونت ، این متغیر هارا میتوانید از لیست متغیر های اماده انتهای صفحه دریافت کنید 

 right مشخصه  

const { style } = require("cls.js");
style({
 text : "متن من",
 right : 10
});

مشخصه راست ورودی یک عدد دریافت میکند که به تعداد ان اعداد متن را به سمت راست تغییر مکان میدهد برای درک اندازه این واحد ها میتوانید ان را تست بکنید ، ما ان را واحد اسپی مینامیم که بسته به اندازه نمایشگر و سایز فونت این واحد بزرگ و کوچک میشود به طور میانگین ۱ تا ۲ پیکسل است ما در بروزرسانی بعدی این واحد اندازه گیری را شخصی سازی کرده و اسپی را رسما معرفی خواهیم کرد 

 top مشخصه  

const { style } = require("cls.js");
style({
 text : "متن من",
 top : 3
});

مشخصه تاپ ورودی یک عدد دریافت میکند که به تعداد ان اعداد متن را به سمت بالا تغییر مکان میدهد برای درک اندازه این واحد ها میتوانید ان را تست بکنید ، ما ان را واحد اسپی مینامیم که بسته به اندازه نمایشگر و سایز فونت این واحد بزرگ و کوچک میشود به طور میانگین ۲ تا ۳ پیکسل است ما در بروزرسانی بعدی این واحد اندازه گیری را شخصی سازی کرد ه و اسپی را رسما معرفی خواهیم کرد 

Loc کار با زمره 

درباره زمره 

این زمره برای کار با موقعیت ها کاربرد دارد ، این نسخه ای ابتدایی از واحد اندازه گیری اسپی است که بیشتر برای ازمایش این قابلیت ارائه گردیده است

  centerW کارور 

const { style , Loc } = require("cls.js");
style({
 text : "متن من",
 right : Loc.centerW("متن من"),
});

این کارور به شکل افقی موقعیت  وسط صفحه را برمیگرداند حتما بایستی مشخصه تکست را به عنوان ورودی  دقیقا همانند مقدار مشخصه وارد کنید تا موقعیت درست را بازگرداند ، شفاف تر بیان میکنم عددی را بازمیگرداند که دقیقا به صورت افقی عدد وسط صفحه است

  centerH کارور 

const { style , Loc } = require("cls.js");
style({
 text : "متن من",
 top : Loc.centerH(),
});

این کارور به شکل عمودی موقعیت  وسط صفحه را برمیگرداند همچنین نیازی به دادن ورودی ای نیست ، شفاف تر بیان میکنم عددی را بازمیگرداند که دقیقا به صورت عمودی عدد وسط صفحه است

left کارور 

const { style , Loc } = require("cls.js");
style({
 text : "متن من",
 right : Loc.left(),
});

این کارور به شکل افقی موقعیت  سمت چپ صفحه را برمیگرداند همچنین نیازی به دادن ورودی ای نیست ، شفاف تر بیان میکنم عددی را بازمیگرداند که دقیقا به صورت افقی عدد سمت چپ صفحه است

right کارور 

const { style , Loc } = require("cls.js");
style({
 text : "متن من",
 right : Loc.right("متن من"),
});

این کارور به شکل افقی موقعیت  وسط صفحه را برمیگرداند حتما بایستی مشخصه تکست را به عنوان ورودی  دقیقا همانند مقدار مشخصه وارد کنید تا موقعیت درست را بازگرداند ، شفاف تر بیان میکنم عددی را بازمیگرداند که دقیقا به صورت افقی عدد وسط صفحه است
